================================================================================
                    6. BASE DE DATOS (TABLAS INFERIDAS)
================================================================================

─────────────────────────────────────────────────────────────────────────────
TABLA: usuarios
─────────────────────────────────────────────────────────────────────────────
COLUMNAS:
- id: UUID (PK, Supabase Auth)
- email: VARCHAR
- rol: VARCHAR (admin/pastor/lider/miembro/finanzas)
- miembro_id: INT (FK → miembros.id, nullable)
- activo: BOOLEAN (default true)
- created_at: TIMESTAMP

─────────────────────────────────────────────────────────────────────────────
TABLA: miembros
─────────────────────────────────────────────────────────────────────────────
COLUMNAS:
- id: INT (PK, auto-increment)
- nombre: VARCHAR
- carnet: VARCHAR (nullable)
- telefono: VARCHAR (nullable)
- direccion: TEXT (nullable)
- fecha_nacimiento: DATE (nullable)
- fecha_conversion: DATE (nullable)
- bautizado: BOOLEAN (default false)
- asistio_encuentro: BOOLEAN (default false)
- estado: VARCHAR (activo/inactivo/visita)
- foto_url: TEXT (nullable, URL pública de Supabase Storage)
- created_at: TIMESTAMP

─────────────────────────────────────────────────────────────────────────────
TABLA: grupos
─────────────────────────────────────────────────────────────────────────────
COLUMNAS:
- id: INT (PK)
- nombre: VARCHAR
- lugar: VARCHAR
- dia_semana: VARCHAR (lunes-martes-miercoles-jueves-viernes-sabado-domingo)
- hora: VARCHAR
- id_lider: INT (FK → miembros.id, nullable)
- estado: VARCHAR (activo/inactivo)
- created_at: TIMESTAMP

─────────────────────────────────────────────────────────────────────────────
TABLA: grupo_miembros
─────────────────────────────────────────────────────────────────────────────
COLUMNAS:
- id: INT (PK)
- id_grupo: INT (FK → grupos.id)
- id_miembro: INT (FK → miembros.id)
- UNIQUE(id_grupo, id_miembro)

─────────────────────────────────────────────────────────────────────────────
TABLA: cursos
─────────────────────────────────────────────────────────────────────────────
COLUMNAS:
- id: INT (PK)
- nombre: VARCHAR
- aula: VARCHAR
- horas: INT (nullable)
- dia_semana: VARCHAR
- hora: VARCHAR
- id_guia: INT (FK → miembros.id, nullable)
- estado: VARCHAR (activo/inactivo)
- precio_curso: DECIMAL (nullable)
- precio_libro: DECIMAL (nullable)
- created_at: TIMESTAMP

─────────────────────────────────────────────────────────────────────────────
TABLA: inscripciones
─────────────────────────────────────────────────────────────────────────────
COLUMNAS:
- id: INT (PK)
- id_miembro: INT (FK → miembros.id)
- id_curso: INT (FK → cursos.id)
- estado: VARCHAR (activo/completado/retirado)
- fecha_inicio: DATE
- periodo_id: INT (FK → periodos_curso.id, nullable)
- UNIQUE(id_miembro, id_curso, periodo_id)

─────────────────────────────────────────────────────────────────────────────
TABLA: periodos_curso
─────────────────────────────────────────────────────────────────────────────
COLUMNAS:
- id: INT (PK)
- id_curso: INT (FK → cursos.id)
- nombre: VARCHAR (ej: "Marzo 2026", "Período 1-2026")
- fecha_fin: DATE
- total_inscritos: INT
- total_completados: INT
- created_at: TIMESTAMP

─────────────────────────────────────────────────────────────────────────────
TABLA: ministerios
─────────────────────────────────────────────────────────────────────────────
COLUMNAS:
- id: INT (PK)
- nombre: VARCHAR
- descripcion: TEXT (nullable)
- estado: VARCHAR (activo/inactivo)
- created_at: TIMESTAMP

─────────────────────────────────────────────────────────────────────────────
TABLA: ministerio_miembros
─────────────────────────────────────────────────────────────────────────────
COLUMNAS:
- id: INT (PK)
- id_ministerio: INT (FK → ministerios.id)
- id_miembro: INT (FK → miembros.id)
- rol_ministerio: VARCHAR (lider/miembro/coordenador, etc.)
- UNIQUE(id_ministerio, id_miembro)
